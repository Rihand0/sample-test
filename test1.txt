%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* ---- support functions ---- */
int nextstat;
int tempcount = 0;
FILE *outfile;

char* newtemp() {
    char *buf = malloc(10);
    sprintf(buf, "t%d", ++tempcount);
    return buf;
}

void emit(char *s) {
    fprintf(outfile, "%d: %s\n", nextstat++, s);
}

int yylex(void);
void yyerror(char *s) { fprintf(stderr, "Error: %s\n", s); }
extern char *yytext;
%}

/* token declarations */
%union {
    char *sval;   /* for ID names and place attributes */
}

%token <sval> ID
%token TRUE FALSE
%token AND OR NOT
%token LT EQ
%token PLUS TIMES
%token ASSIGN

%type <sval> B E

/* precedence — lowest to highest */
%right ASSIGN
%left OR AND
%left LT EQ
%left PLUS
%left TIMES
%right NOT

%%

program :
    S
    ;

S :
    ID ASSIGN B ';'
    {
        char buf[100];
        sprintf(buf, "%s := %s", $1, $3);
        emit(buf);
        fprintf(outfile, "Assignment statement processed\n");
    }
    ;

B :
    TRUE
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := 1", t);
        emit(buf);
        $$ = t;
    }
    | FALSE
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := 0", t);
        emit(buf);
        $$ = t;
    }
    | B OR B
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := %s OR %s", t, $1, $3);
        emit(buf);
        $$ = t;
    }
    | B AND B
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := %s AND %s", t, $1, $3);
        emit(buf);
        $$ = t;
    }
    | NOT B
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := NOT %s", t, $2);
        emit(buf);
        $$ = t;
    }
    | '(' B ')'
    {
        $$ = $2;
    }
    | B EQ B
    {
        /* pattern: if E1==E2 goto next+3
                    t := 0
                    goto next+2
                    t := 1        */
        char *t = newtemp();
        char buf[100];
        int tgt_true  = nextstat + 3;
        int tgt_after = nextstat + 4;

        sprintf(buf, "if %s==%s goto %d", $1, $3, tgt_true);
        emit(buf);                          /* label N   */

        sprintf(buf, "%s := 0", t);
        emit(buf);                          /* label N+1 */

        sprintf(buf, "goto %d", tgt_after);
        emit(buf);                          /* label N+2 */

        sprintf(buf, "%s := 1", t);
        emit(buf);                          /* label N+3 */

        $$ = t;
    }
    | B LT B
    {
        char *t = newtemp();
        char buf[100];
        int tgt_true  = nextstat + 3;
        int tgt_after = nextstat + 4;

        sprintf(buf, "if %s<%s goto %d", $1, $3, tgt_true);
        emit(buf);

        sprintf(buf, "%s := 0", t);
        emit(buf);

        sprintf(buf, "goto %d", tgt_after);
        emit(buf);

        sprintf(buf, "%s := 1", t);
        emit(buf);

        $$ = t;
    }
    | B PLUS B
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := %s + %s", t, $1, $3);
        emit(buf);
        $$ = t;
    }
    | B TIMES B
    {
        char *t = newtemp();
        char buf[100];
        sprintf(buf, "%s := %s * %s", t, $1, $3);
        emit(buf);
        $$ = t;
    }
    | ID
    {
        $$ = $1;
    }
    ;

%%

int main() {
    char filename[50];
    scanf("%d", &nextstat);   /* read starting label from first line of input */
    outfile = fopen("output.txt", "w");
    yyparse();
    fclose(outfile);
    return 0;
}
